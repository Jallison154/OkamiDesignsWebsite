cd /opt/okami-designs

# 1. Back up live server data
cp files/manifest.json /tmp/manifest.json.backup
cp files/site-settings.json /tmp/site-settings.json.backup
cp files/analytics.json /tmp/analytics.json.backup

# 2. Undo skip-worktree (temporary)
git update-index --no-skip-worktree files/manifest.json 2>/dev/null
git update-index --no-skip-worktree files/site-settings.json 2>/dev/null
git update-index --no-skip-worktree files/analytics.json 2>/dev/null

# 3. Reset tracked files that block the merge (your PDF/PNG are untracked — safe)
git checkout -- deploy.sh files/manifest.json files/site-settings.json files/analytics.json

# 4. Pull — should fast-forward 8 commits now
git pull origin main

# 5. Restore production data
cp /tmp/manifest.json.backup files/manifest.json
cp /tmp/site-settings.json.backup files/site-settings.json
cp /tmp/analytics.json.backup files/analytics.json

# 6. Tell git to ignore local edits on next pulls
git update-index --skip-worktree files/manifest.json
git update-index --skip-worktree files/site-settings.json
git update-index --skip-worktree files/analytics.json

# 7. Deploy
chmod +x deploy.sh
./deploy.sh
